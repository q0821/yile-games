# 連六棋（Connect6）Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans（本次由主 session inline 執行）。Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在弈樂新增可玩的「連六棋」棋種（Web/Android/iOS 三版），標準規則、pvc+pvp、貪婪雙落子 AI、整回合悔棋。

**Architecture:** 沿用「每種棋自成 `X-rules/ai/mode/ui.js`」慣例（取向 A1），複製五子棋骨架改成連六棋；純函式（`candidates` 等）import 重用，五子棋碼不動。回合狀態機以 `pending`（本回合未提交的子）＋ `history`（每回合一筆）驅動「每手兩子」。

**Tech Stack:** 原生 ES module + Canvas；Jest（vm sandbox，見 `tests/helpers.js`）。

## Global Constraints

- `SIZE=19`、`WIN_LEN=6`；連 6 子（含以上）勝，長連也算贏。
- 回合結構：整局第一回合（黑）下 1 子，其餘每回合 2 子。
- 座標慣例 `board[row][col]`（沿用全專案）。
- 卡片標語：`雙落連橫，六子成龍`；id=`connect6`、hash=`#connect6`、title=`連六棋`。
- **不掛 `webOnly`**（iOS 納入）；**不加入** `IOS_EXCLUDED_HASHES`。
- UI 不放 emoji。commit 訊息不得含 Claude 署名。
- 收尾前 bump `public/version.json` 與 `public/sw.js` cache-bust 版號。

```js
// connect6-rules.js — 連六棋純規則（無 DOM，可單元測試）。
// 19 路盤、每點空即可下、任一方向連成 6 子（含以上，長連也算）即勝。
import { EMPTY, createBoard, inBounds } from './rules.js';
import { canPlace as gomokuCanPlace, isBoardFull as gomokuIsBoardFull } from './gomoku-rules.js';

export const SIZE = 19;
export const WIN_LEN = 6;

const DIRS = [[0, 1], [1, 0], [1, 1], [1, -1]];

export function newBoard(size = SIZE) { return createBoard(size); }
export const canPlace = gomokuCanPlace;      // 界內且為空 — 與五子棋同義，直接重用
export const isBoardFull = gomokuIsBoardFull; // 與五子棋同義，直接重用

export function checkWin(board, size, r, c, player) {
  if (!inBounds(size, r, c) || board[r][c] !== player) return { won: false, line: [] };
  for (const [dr, dc] of DIRS) {
    const cells = [[r, c]];
    let nr = r + dr, nc = c + dc;
    while (inBounds(size, nr, nc) && board[nr][nc] === player) { cells.push([nr, nc]); nr += dr; nc += dc; }
    nr = r - dr; nc = c - dc;
    while (inBounds(size, nr, nc) && board[nr][nc] === player) { cells.unshift([nr, nc]); nr -= dr; nc -= dc; }
    if (cells.length >= WIN_LEN) return { won: true, line: cells };
  }
  return { won: false, line: [] };
}

export const Connect6Rules = { SIZE, WIN_LEN, newBoard, canPlace, checkWin, isBoardFull };
```

```js
// connect6-ai.js — 連六棋 AI（V1 貪婪雙落子；純邏輯、無 DOM）。
// 每回合依序選 1~2 子：致勝 → 擋對手立即六 → 啟發式評分。候選點與五子棋共用 candidates()。
import { EMPTY, opponent, inBounds } from './rules.js';
import { checkWin } from './connect6-rules.js';
import { candidates } from './gomoku-ai.js';

const DIRS = [[0, 1], [1, 0], [1, 1], [1, -1]];

// 6 門檻型態分值：連 5（有開放端）＝迫著、活四次之。數字只需相對大小正確。
function patternValue(count, openEnds) {
  if (count >= 6) return 1e8;
  if (count === 5) return openEnds >= 1 ? 1e6 : 0;
  if (count === 4) return openEnds === 2 ? 5e4 : openEnds === 1 ? 5e3 : 0;
  if (count === 3) return openEnds === 2 ? 1e3 : openEnds === 1 ? 100 : 0;
  if (count === 2) return openEnds === 2 ? 100 : openEnds === 1 ? 10 : 0;
  if (count === 1) return openEnds === 2 ? 10 : openEnds === 1 ? 2 : 0;
  return 0;
}

function dirCount(board, size, r, c, player, dr, dc) {
  let count = 1, openEnds = 0;
  let nr = r + dr, nc = c + dc;
  while (inBounds(size, nr, nc) && board[nr][nc] === player) { count++; nr += dr; nc += dc; }
  if (inBounds(size, nr, nc) && board[nr][nc] === EMPTY) openEnds++;
  nr = r - dr; nc = c - dc;
  while (inBounds(size, nr, nc) && board[nr][nc] === player) { count++; nr -= dr; nc -= dc; }
  if (inBounds(size, nr, nc) && board[nr][nc] === EMPTY) openEnds++;
  return { count, openEnds };
}

export function placeScore(board, size, r, c, player) {
  board[r][c] = player;
  let s = 0;
  for (const [dr, dc] of DIRS) {
    const { count, openEnds } = dirCount(board, size, r, c, player, dr, dc);
    s += patternValue(count, openEnds);
  }
  board[r][c] = EMPTY;
  return s;
}

/** 在 work 盤上選「一子」（會被 bestTurn 連呼兩次；work 已含前一子）。 */
function bestOne(board, size, player, level, rng) {
  const opp = opponent(player);
  const cands = candidates(board, size);
  if (!cands.length) {
    const mid = (size / 2) | 0;
    return board[mid][mid] === EMPTY ? { r: mid, c: mid } : null;
  }
  // a) 立即致勝
  for (const { r, c } of cands) {
    board[r][c] = player;
    const won = checkWin(board, size, r, c, player).won;
    board[r][c] = EMPTY;
    if (won) return { r, c };
  }
  // b) 擋對手立即致勝
  for (const { r, c } of cands) {
    board[r][c] = opp;
    const won = checkWin(board, size, r, c, opp).won;
    board[r][c] = EMPTY;
    if (won) return { r, c };
  }
  // c) 啟發式：攻 + 守
  const scored = cands.map(({ r, c }) => {
    const attack = placeScore(board, size, r, c, player);
    const defend = placeScore(board, size, r, c, opp);
    let score = attack + defend * 0.95;
    if (level >= 3) {
      board[r][c] = player;
      const reply = candidates(board, size).reduce((mx, m) =>
        Math.max(mx, placeScore(board, size, m.r, m.c, opp)), 0);
      board[r][c] = EMPTY;
      score -= reply * 0.8;
    }
    return { r, c, score };
  });
  scored.sort((a, b) => b.score - a.score);
  if (level <= 1) {
    const k = Math.min(4, scored.length);
    return scored[Math.floor(rng() * k)];
  }
  const top = scored.filter((s) => s.score === scored[0].score);
  return top[Math.floor(rng() * top.length)];
}

/** 回傳這回合要下的 1~2 子（quota 決定；已致勝則提早結束）。 */
export function bestTurn(board, size, player, level = 2, quota = 2, rng = Math.random) {
  const work = board.map((row) => row.slice());
  const moves = [];
  for (let i = 0; i < quota; i++) {
    const m = bestOne(work, size, player, level, rng);
    if (!m) break;
    work[m.r][m.c] = player;
    moves.push({ r: m.r, c: m.c });
    if (checkWin(work, size, m.r, m.c, player).won) break;
  }
  return moves;
}

export const Connect6AI = { placeScore, bestTurn };
```
